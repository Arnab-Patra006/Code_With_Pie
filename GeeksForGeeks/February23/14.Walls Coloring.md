class Solution{   
public:
int ans;
int cnt;
void dfs(vector<vector<int>>&colors,int N,int element,int color,int cost)
{
    // cnt++;
    // cout<<element<<","<<color<<"-->"<<cost<<endl;
    if(element==N-1)
    {
        cout<<cost<<endl;
        ans=min(ans,cost);
        return;
    }
    int curr_cost=cost+colors[element][color];
    if(color==0)
    {
        dfs(colors,N,element+1,1,curr_cost);
        dfs(colors,N,element+1,2,curr_cost);
    }
    else if(color==1)
    {
        dfs(colors,N,element+1,0,curr_cost);
        dfs(colors,N,element+1,2,curr_cost);
    }
    else if(color==2)
    {
        dfs(colors,N,element+1,0,curr_cost);
        dfs(colors,N,element+1,1,curr_cost);
    }
    
    return;
}
    int minCost(vector<vector<int>> &colors, int N) {
        // Write your code here.
        ans=INT_MAX;
        cnt=0;
        for(int i=0;i<N;i++)
        {
            for(int j=0;j<3;j++)
            {
                dfs(colors,N,i,j,0);
            }
        }
        // cout<<"==>"<<ans<<endl;
        // cout<<cnt<<endl;
        return ans;
    }
};
